version: '3.3'
services:
    alist:
        image: 'xhofe/alist-aria2:latest'
        volumes:
            - './data:/opt/alist/data'
            - '/home:/home'
            - '/root:/root'
        ports:
            - '5246:5244'
        #environment:
            #- PUID=0
            #- PGID=0
            #- UMASK=022
        container_name: alist
        restart: unless-stopped
        networks:
            - alist_network

networks:
    alist_network:
        external: true

1
